buat folder view di masing masing nama projek  dn buat file views.py dan masukan script ini 
from django.shortcuts import render
from django.views import View


class IndexView(View):
    def get(self, request):
        # arahkan return ini ke path html yang sudah kita buat tadi tanpa folder tampatels jadi langsung ke folder/file yang berada di dalam tampates.
        return render(request, 'app/index.html')
